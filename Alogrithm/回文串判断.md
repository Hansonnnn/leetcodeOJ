 public boolean isPalindrome(int x) {
        if(x<0){
            return false;
        }
        int y=x;
        int result=0;
        while(x!=0){
            int tail=x%10;
            int newResult=result*10+tail;
            if((newResult-tail)/10!=result){
                return false;
            }
            result=newResult;
            x=x/10;
        }
        if(result==y){
            return true;
        }else{
            return false;
        }
}
